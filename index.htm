💡In Node.js, process.nextTick runs before Promise microtasks. It's technically a separate queue with higher priority. This catches people out when mixing nextTick with Promise.then.

02 Closures, Scope & Hoisting Mechanics
Scope Chain
function outer() {
  const x = 10;
  function inner() {
    // x captured via closure
    // Lives in outer's [[Env]]
    return x * 2;
  }
  return inner;
}
const fn = outer();
fn(); // 20 — outer is GC'd except x
Hoisting Rules
// var → hoisted + initialized undefined
console.log(a); // undefined
var a = 1;

// let/const → hoisted, NOT initialized
console.log(b); // ReferenceError
let b = 2; // TDZ above this line

// function declarations → fully hoisted
greet(); // Works!
function greet() {};
Classic Closure Pitfall — Loop with var
// ❌ Wrong — all callbacks share same 'i'
for (var i = 0; i < 3; i++) {
  setTimeout(() => console.log(i), 0); // 3, 3, 3
}

// ✅ Fix 1: let (block-scoped per iteration)
for (let i = 0; i < 3; i++) {
  setTimeout(() => console.log(i), 0); // 0, 1, 2
}

// ✅ Fix 2: IIFE (creates new scope each iteration)
for (var i = 0; i < 3; i++) {
  ((j) => setTimeout(() => console.log(j), 0))(i);
}

  const results = [];

Async/Await Patterns & Pitfalls
// ❌ Sequential — 300ms total (avoidable!)
const a = await fetchA(); // 100ms
const b = await fetchB(); // 100ms
const c = await fetchC(); // 100ms

// ✅ Parallel — ~100ms total
const [a, b, c] = await Promise.all([fetchA(), fetchB(), fetchC()]);

// ❌ Missing await on async forEach
items.forEach(async (item) => {
  await save(item); // forEach doesn't await async callbacks!
});

// ✅ Use for...of for sequential async
for (const item of items) {
  await save(item);
}
Async Generators — Streaming / Pagination
async function* paginate(url) {
  let cursor = null;
  do {
    const res = await fetch(`${url}?cursor=${cursor}`).then(r => r.json());
    yield res.data;
    cursor = res.nextCursor;
  } while (cursor);
}

for await (const page of paginate('/api/users')) {
  await processPage(page); // Memory-efficient: one page at a time
}
💡Async generators are excellent for streaming large datasets from DBs or external APIs without loading everything into memory. Perfect for ETL pipelines.

Proxy & Reflect — Metaprogramming
// Validation proxy
function createValidator(target, schema) {
  return new Proxy(target, {
    set(obj, prop, value) {
      if (schema[prop] && !schema[prop].test(value))
        throw new TypeError(`Invalid value for ${prop}`);
      return Reflect.set(obj, prop, value);
    }
  });
}

Type Coercion

null === undefined // false
NaN === NaN // false!

Float Precision
0.1 + 0.2 // 0.30000000000000004

// Fix: integer math
(0.1*10 + 0.2*10) / 10 // 0.3

// Or: toFixed + parseFloat
parseFloat((0.1+0.2).toFixed(10))

// For money: use BigInt or decimal.js
'this' Binding
class Timer {
  constructor() {
    this.count = 0;
    // ❌ 'this' lost in callback
    setInterval(function() {
      this.count++; // undefined.count
    }, 1000);
    // ✅ Arrow inherits 'this'
    setInterval(() => this.count++, 1000);
  }
}
JSON & Date Gotchas
// JSON.stringify loses: undefined, functions, symbols, BigInt
JSON.stringify({ a: undefined, b: () => {}, c: 42n });
// → '{}' — silently drops a and b, throws on c

// Date round-trip: JSON serializes to ISO string, not Date object
const orig = { date: new Date() };
const parsed = JSON.parse(JSON.stringify(orig));
parsed.date instanceof Date; // false — it's a string!

// Fix: use reviver
JSON.parse(str, (key, val) =>
  typeof val === 'string' && ISO_RE.test(val) ? new Date(val) : val
);
⚠️Array.sort() is not stable in all engines before Node 12 — avoid sorting objects by multiple fields without a deterministic comparator. As of V8 7.0+ (Node 12+), sort is stable (TimSort). Always use an explicit comparator: (a, b) => a.age - b.age || a.name.localeCompare(b.name)
Symbol — Unique Keys & Well-Known Symbols
// Custom iterator — implement Symbol.iterator
class Range {
  constructor(start, end) { this.start = start; this.end = end; }
  [Symbol.iterator]() {
    let current = this.start;
    const end = this.end;
    return {
      next() {
        return current <= end
          ? { value: current++, done: false }
          : { done: true };
      }
    };
  }
}
for (const n of new Range(1, 5)) console.log(n); // 1 2 3 4 5
[...new Range(1, 3)]; // [1, 2, 3]
